# JavaScriptで電卓を作ってほしい

以下は単一のHTMLファイルとして保存して実行できる電卓のソースコードです。コピーして `index.html` として保存し、ブラウザで開いてください。

```html
<!doctype html>
<html lang="ja">
<head>
	<meta charset="utf-8" />
	<meta name="viewport" content="width=device-width,initial-scale=1" />
	<title>電卓</title>
	<style>
		:root{--bg:#f4f7fb;--panel:#fff;--accent:#4a90e2;--btn:#e6eefc}
		html,body{height:100%;margin:0;font-family:system-ui, -apple-system, "Segoe UI", Roboto, "Hiragino Kaku Gothic ProN", "Meiryo", sans-serif;background:var(--bg);display:flex;align-items:center;justify-content:center}
		.calculator{width:320px;background:var(--panel);border-radius:12px;box-shadow:0 8px 24px rgba(10,20,40,0.08);padding:18px}
		#display{width:100%;height:56px;border-radius:8px;border:1px solid #e6eefc;padding:12px;font-size:24px;text-align:right;box-sizing:border-box;background:#f8fbff}
		.buttons{display:grid;grid-template-columns:repeat(4,1fr);gap:10px;margin-top:12px}
		button{height:52px;border:0;border-radius:8px;background:var(--btn);font-size:18px;cursor:pointer}
		button.op{background:#fff;color:#333;border:1px solid #e6eefc}
		button.equal{background:var(--accent);color:#fff;font-weight:600}
		button.span2{grid-column:span 2}
	</style>
</head>
<body>
	<div class="calculator">
		<input id="display" type="text" readonly value="0" />
		<div class="buttons">
			<button data-action="clear">C</button>
			<button data-action="back">←</button>
			<button data-value="%">%</button>
			<button data-value="/" class="op">÷</button>

			<button data-value="7">7</button>
			<button data-value="8">8</button>
			<button data-value="9">9</button>
			<button data-value="*" class="op">×</button>

			<button data-value="4">4</button>
			<button data-value="5">5</button>
			<button data-value="6">6</button>
			<button data-value="-" class="op">−</button>

			<button data-value="1">1</button>
			<button data-value="2">2</button>
			<button data-value="3">3</button>
			<button data-value="+" class="op">+</button>

			<button data-value="0" class="span2">0</button>
			<button data-value="." class="op">.</button>
			<button data-action="equals" class="equal">=</button>
		</div>
	</div>

	<script>
		(()=>{
			const display = document.getElementById('display')
			let expr = ''

			function update(v){
				if(expr === '' && v === '0') return setDisplay('0')
				expr += v
				setDisplay(expr)
			}

			function setDisplay(v){ display.value = String(v) }

			function clearAll(){ expr=''; setDisplay('0') }

			function backspace(){ expr = expr.slice(0,-1); setDisplay(expr||'0') }

			function compute(){
				try{
					// 安全のため、許可する文字だけを残す
					const safe = expr.replace(/[^0-9.+\\-*/()%]/g,'')
					// Functionで評価
					const result = Function('return ('+safe+')')()
					expr = String(result)
					setDisplay(expr)
				}catch(e){ setDisplay('Error'); expr=''}
			}

			document.querySelectorAll('.buttons button').forEach(b=>{
				b.addEventListener('click',()=>{
					const v = b.getAttribute('data-value')
					const a = b.getAttribute('data-action')
					if(a==='clear'){ clearAll(); return }
					if(a==='back'){ backspace(); return }
					if(a==='equals'){ compute(); return }
					if(v){
						// パーセント処理：%を入力すると/100で扱う
						if(v === '%'){
							expr = String((parseFloat(expr)||0)/100)
							setDisplay(expr)
						}else{
							update(v)
						}
					}
				})
			})
		})()
	</script>
</body>
</html>
```
