# JavaScriptデジタル時計 - Proof of Mockup

## プロジェクト概要
JavaScriptを使用して、リアルタイムで現在時刻を表示するデジタル時計を作成する。

## 機能要件

### 基本機能
1. **リアルタイム表示**
   - 現在の時刻を「HH:MM:SS」形式で表示
   - 1秒ごとに自動更新

2. **時刻表示フォーマット**
   - 24時間表記
   - 常に2桁表示（例：09:05:03）

3. **日付表示（オプション）**
   - 現在の日付を表示
   - 曜日も含める

4. **複数国タイムゾーン表示**
   - 日本時刻を主要表示（大きいサイズ）
   - 主要国の時刻を小さいサイズで並行表示
   - 全国の時刻が動画面でリアルタイム同期更新
   - 対応国家：ニューヨーク、ロンドン、パリ、シドニー、ドバイ、シンガポール、バンコク

### ユーザーインターフェース
- 日本時刻を中央に大きく表示（フォントサイズ120px）
- 他国の時刻を小さいサイズで表示（フォントサイズ40px）
- グリッド状レイアウトで複数国を配置
- 国名とタイムゾーンを表示
- 黒背景に白テキストの高コントラスト配色
- レスポンシブデザイン対応

#### メイン更新関数
```javascript
/**
 * 指定されたタイムゾーンの現在時刻を取得
 * @param {string} timezone - タイムゾーン（例：'Asia/Tokyo'）
 * @returns {object} {hours, minutes, seconds, year, month, date, dayOfWeek}
 */
function getTimeInTimezone(timezone) {
  const now = new Date();
  
  // Intl.DateTimeFormatを使ってタイムゾーン対応の日時を取得
  const formatter = new Intl.DateTimeFormat('ja-JP', {
    timeZone: timezone,
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit',
    weekday: 'narrow'
  });
  
  const parts = formatter.formatToParts(now);
  const result = {};
  
  parts.forEach(part => {
    result[part.type] = part.value;
  });
  
  return {
    hours: result.hour,
    minutes: result.minute,
    seconds: result.second,
    year: result.year,
    month: result.month,
    date: result.day,
    dayOfWeek: result.weekday
  };
}

/**
 * 曜日の日本語名に変換
 * @param {string} day - 曜日（例：'月'）
 * @returns {string} 日本語曜日
 */
function getJapaneseDayOfWeek(day) {
  const dayMap = {
    '日': '日', '月': '月', '火': '火', '水': '水',
    '木': '木', '金': '金', '土': '土'
  };
  return dayMap[day] || day;
}

/**
 * 全てのクロック表示を更新
 */
function updateAllClocks() {
  // 時刻表示要素を全て取得
  const timeElements = document.querySelectorAll('.time[data-timezone]');
  const dateElements = document.querySelectorAll('.date[data-timezone]');
  
  timeElements.forEach(element => {
    const timezone = element.getAttribute('data-timezone');
    const timeData = getTimeInTimezone(timezone);
    
    // 時刻フォーマット：HH:MM:SS
    const timeString = `${timeData.hours}:${timeData.minutes}:${timeData.seconds}`;
    element.textContent = timeString;
  });
  
  // 日付表示要素を更新（日本のみ）
  const japanDateElement = document.querySelector('.date[data-timezone="Asia/Tokyo"]');
  if (japanDateElement) {
    const timeData = getTimeInTimezone('Asia/Tokyo');
    const dayOfWeek = getJapaneseDayOfWeek(timeData.dayOfWeek);
    const dateString = `${timeData.year}年${timeData.month}月${timeData.date}日（${dayOfWeek}）`;
    japanDateElement.textContent = dateString;
  }
}

/**
 * 初期化と更新開始
 */
document.addEventListener('DOMContentLoaded', function() {
  // 初期表示
  updateAllClocks();
  
  // 1秒ごとに更新
  setInterval(updateAllClocks, 1000);
});
```
